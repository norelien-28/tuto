    }

7. Service mÃ©tier : src/services/UtilisateurService.php

namespace App\Services;

class UtilisateurService {
    private $pdo;

    public function __construct() {
        $this->pdo = require __DIR__ . '/../database/connection.php';
    }

    public function getAll() {
        $stmt = $this->pdo->query("SELECT * FROM utilisateurs");
        return $stmt->fetchAll(\PDO::FETCH_ASSOC);
    }

    public function create($data) {
        $stmt = $this->pdo->prepare("INSERT INTO utilisateurs (nom) VALUES (?)");
        $stmt->execute([$data['nom']]);
        return ['id' => $this->pdo->lastInsertId(), 'nom' => $data['nom']];
    }
}
