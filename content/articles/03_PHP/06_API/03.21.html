4. Exemple de test simple avec Guzzle (API en local)

Fichier : tests/UtilisateurApiTest.php

php
<?php

use PHPUnit\Framework\TestCase;
use GuzzleHttp\Client;

class UtilisateurApiTest extends TestCase
{
    protected Client $client;

    protected function setUp(): void
    {
        $this->client = new Client([
            'base_uri' => 'http://localhost:8000/api/',
            'http_errors' => false // Ne pas throw les erreurs 4xx/5xx
        ]);
    }

    public function testGetUtilisateurs()
    {
        $response = $this->client->get('utilisateurs');
        $this->assertEquals(200, $response->getStatusCode());

        $data = json_decode($response->getBody(), true);
        $this->assertIsArray($data);
    }

    public function testCreateUtilisateur()
    {
        $response = $this->client->post('utilisateurs', [
            'form_params' => ['nom' => 'TestNom']
        ]);

        $this->assertEquals(201, $response->getStatusCode());

        $data = json_decode($response->getBody(), true);
        $this->assertArrayHasKey('id', $data);
        $this->assertEquals('TestNom', $data['nom']);
    }
}
